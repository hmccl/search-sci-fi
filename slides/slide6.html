Apresentação do Buscador de Filmes

Interface

  • Recebe o link inicial
  • Busca enquanto o usuário estiver interagido
// interação com o usuário
async function askUser(question) {
  return new Promise((resolve) => {
    rl.question(question, (answer) => {
      resolve(answer.trim());
    });
  });
}

async function main() {
  // URL inicial
  let urlPage = await askUser('Endereço da página inicial:\n');
  await crawlPage(urlPage);

  try {
    // loop de termo de busca
    while (true) {
      let searchTerm = await askUser('Termo de busca:\n');
      await searchPage(searchTerm);
      await displayResults();
    }
  } catch (error) {
    console.error('Erro na execução\n');
  }
}
